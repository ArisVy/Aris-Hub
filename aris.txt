-- 2. ĐIỀU KHIỂN NÚT BẤM
-- ==========================================
ToggleMenuBtn.MouseButton1Click:Connect(function()
    MenuFrame.Visible = not MenuFrame.Visible
end)

local isAttacking = false
AttackBtn.MouseButton1Click:Connect(function()
    isAttacking = not isAttacking
    AttackBtn.Text = isAttacking and "FAST M1: ON" or "FAST M1: OFF"
    AttackBtn.BackgroundColor3 = isAttacking and Color3.fromRGB(50, 200, 50) or Color3.fromRGB(200, 50, 50)
end)

-- ==========================================
-- 3. LOGIC FAST M1 CHUẨN TỪ FILE SOURCE
-- (Trích xuất từ hàm FruitAttackPlayer và T-Rex Combat Loop)
-- ==========================================
task.spawn(function()
    while true do
        -- Độ trễ y hệt trong vòng lặp M1 của file source
        task.wait(0.01) 
        
        if isAttacking then
            pcall(function()
                local char = lp.Character
                if not char then return end
                
                -- Tìm vũ khí ĐANG CẦM (Không hardcode T-Rex)
                local tool = char:FindFirstChildOfClass("Tool")
                if tool then
                    -- Lấy Remote giống cách file source gọi: tool:FindFirstChild(config.RemoteName)
                    -- (Trong file config.RemoteName đa số là "LeftClickRemote")
                    local remote = tool:FindFirstChild("LeftClickRemote") or tool:FindFirstChild("RemoteFunction") or tool:FindFirstChild("RemoteEvent")
                    
                    if remote then
                        -- Tọa độ hướng đánh (Hardcode Vector3.new(0, -0.9, 0.03) lấy trực tiếp từ hàm FruitAttackPlayer trong file)
                        local direction = Vector3.new(0, -0.9, 0.03)
                        
                        -- Nếu là RemoteEvent (Trái ác quỷ / Súng thường dùng cái này)
                        if remote:IsA("RemoteEvent") then
                            -- Bắn thẳng lên server, số 1 là tham số args combo (giống Kitsune/Dragon trong file)
                            remote:FireServer(Vector3.new(direction.X, direction.Y, direction.Z), 1, true)
                        
                        -- Nếu là RemoteFunction (Melee / Kiếm thường dùng cái này)
                        elseif remote:IsA("RemoteFunction") then
                            remote:InvokeServer()
                        end
                    end
                end
            end)
        end
    end
end)
