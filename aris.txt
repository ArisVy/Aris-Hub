task.spawn(function()
    while true do
        task.wait(0.01) 
        
        if isAttacking then
            pcall(function()
                local char = lp.Character
                if not char then return end
                
                -- ==========================================
                -- TÍNH NĂNG TĂNG RANGE (HITBOX EXPANDER)
                -- ==========================================
                local RANGE_SIZE = 60 -- Chỉnh kích thước tầm đánh ở đây (60 là khoảng cách an toàn)
                
                -- Giả sử quái nằm trong workspace.Enemies (thư mục chuẩn của Blox Fruits)
                -- Nếu đánh người thì thay bằng game:GetService("Players"):GetPlayers()
                if workspace:FindFirstChild("Enemies") then
                    for _, enemy in pairs(workspace.Enemies:GetChildren()) do
                        local rootPart = enemy:FindFirstChild("HumanoidRootPart")
                        local humanoid = enemy:FindFirstChild("Humanoid")
                        
                        -- Nếu quái còn sống và có RootPart
                        if rootPart and humanoid and humanoid.Health > 0 then
                            -- Phóng to hitbox của quái
                            rootPart.Size = Vector3.new(RANGE_SIZE, RANGE_SIZE, RANGE_SIZE)
                            rootPart.Transparency = 0.9 -- Làm trong suốt để đỡ che màn hình
                            rootPart.CanCollide = false -- Tắt va chạm để nhân vật không bị kẹt vào quái
                        end
                    end
                end
                
                -- ==========================================
                -- LOGIC ĐÁNH (GỮI NGUYÊN BẢN CỦA BẠN)
                -- ==========================================
                local tool = char:FindFirstChildOfClass("Tool")
                if tool then
                    local remote = tool:FindFirstChild("LeftClickRemote") or tool:FindFirstChild("RemoteFunction") or tool:FindFirstChild("RemoteEvent")
                    
                    if remote then
                        local direction = Vector3.new(0, -0.9, 0.03)
                        if remote:IsA("RemoteEvent") then
                            remote:FireServer(Vector3.new(direction.X, direction.Y, direction.Z), 1)
                        elseif remote:IsA("RemoteFunction") then
                            remote:InvokeServer()
                        end
                    end
                end
            end)
        end
    end
end)
