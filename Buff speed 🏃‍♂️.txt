-- Draggable helper
local function makeDraggable(element)
    local dragging, dragInput, dragStart, startPos
    
    element.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = element.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    element.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            element.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)
end

-- Logic điều khiển
-- ==========================================
local minSpeed = 16
local maxSpeed = 250

local function updateSlider(value)
    local clamped = math.clamp(value, minSpeed, maxSpeed)
    local ratio = (clamped - minSpeed) / (maxSpeed - minSpeed)
    sliderBar.Size = UDim2.new(ratio, 0, 1, 0)
    sliderKnob.Position = UDim2.new(ratio, -10, 0.5, -10)
    valueLabel.Text = "Speed: " .. math.floor(clamped)
    _G.WalkSpeed = math.floor(clamped)
end

updateSlider(_G.WalkSpeed)  -- init tại 90

local draggingSlider = false

sliderFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        draggingSlider = true
    end
end)

sliderFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        draggingSlider = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if draggingSlider and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local mousePos = input.Position
        local sliderAbsPos = sliderFrame.AbsolutePosition
        local sliderAbsSize = sliderFrame.AbsoluteSize
        
        local relativeX = math.clamp((mousePos.X - sliderAbsPos.X) / sliderAbsSize.X, 0, 1)
        local value = minSpeed + relativeX * (maxSpeed - minSpeed)
        updateSlider(value)
    end
end)

-- Nút +/- logic
minus10.MouseButton1Click:Connect(function() updateSlider(_G.WalkSpeed - 10) end)
minus5.MouseButton1Click:Connect(function() updateSlider(_G.WalkSpeed - 5) end)
plus5.MouseButton1Click:Connect(function() updateSlider(_G.WalkSpeed + 5) end)
plus10.MouseButton1Click:Connect(function() updateSlider(_G.WalkSpeed + 10) end)

-- Toggle logic
toggleButton.MouseButton1Click:Connect(function()
    _G.WalkSpeedEnabled = not _G.WalkSpeedEnabled
    
    if _G.WalkSpeedEnabled then
        toggleButton.Text = "ENABLED: ON"
        toggleButton.BackgroundColor3 = Color3.fromRGB(40, 120, 40)
        ToggleBtn.Text = "ON"
        ToggleBtn.TextColor3 = Color3.fromRGB(100, 255, 100)
    else
        toggleButton.Text = "ENABLED: OFF"
        toggleButton.BackgroundColor3 = Color3.fromRGB(80, 40, 40)
        ToggleBtn.Text = "S"
        ToggleBtn.TextColor3 = Color3.fromRGB(255,255,255)
    end
    
    StarterGui:SetCore("SendNotification", {
        Title = "WalkSpeed",
        Text = _G.WalkSpeedEnabled and ("BẬT - " .. _G.WalkSpeed) or "TẮT",
        Duration = 2
    })
end)

-- Mở/đóng menu

-- ÁP DỤNG WALKSPEED

    for _, v in ipairs(RainbowList) do

    end

    if LocalPlayer.Character then

        if hum then
            hum.WalkSpeed = _G.WalkSpeedEnabled and _G.WalkSpeed or 16
        end
    end
end)
